class CombinedAttributesAdder(BaseEstimator, TransformerMixin):
    def __init__(self):
        self.total_rooms_i = 3
        self.total_bedrooms_i = 4
        self.households_i = 6
    
    def fit(self, X, y=None):
        return self
    
    def transform(self, X, y=None):
        total_rooms_per_hhold = \
        X[:, self.total_rooms_i] / X[:, self.households_i]
        total_bedrooms_per_hhold = \
        X[:, self.total_bedrooms_i] / X[:, self.households_i]
        
        return np.c_[
            X, 
            total_rooms_per_hhold, 
            total_bedrooms_per_hhold
        ]
